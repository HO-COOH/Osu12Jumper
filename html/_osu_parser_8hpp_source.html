48 static inline auto SplitKeyVal(std::string_view line, char separator = ':')
49 {
50 if (auto const splitPos = line.find(separator); splitPos != std::string::npos)
51 {
52 auto const valuePos = line.find_first_not_of(" ", splitPos + 1);
53 return std::pair{
54 splitPos == 0 ? "" : line.substr(0, line.find_last_not_of(" ", splitPos - 1) + 1),
55 valuePos != std::string::npos ? line.substr(valuePos) : ""
56 };
57 }
58
59 return std::pair<std::string_view, std::string_view>{"", ""};
60 }

69 static inline auto SplitCommaSeperatedString(std::string_view s)
70 {
71 std::vector<int> result;
72 std::stringstream ss{ s.data() };
73
74 for (int i{}; ss >> i; )
75 {
76 result.push_back(i);
77 if (ss.peek() == ',' || ss.peek() == ' ')
78 ss.ignore();
79 }
80
81 return result;
82 }

84 template<size_t N>
85 static inline auto SplitString(std::string_view str, char delim = ',', bool appendRest = false)
86 {
87 std::array<std::string_view, N> result;
88 size_t startPos = 0, endPos = str.find(delim);
89 size_t i = 0;
90 while (endPos != std::string::npos && i < N)
91 {
92 if (appendRest && i + 1 == N)
93 result[i] = str.substr(startPos);
94 else
95 result[i] = str.substr(startPos, endPos - startPos);
96 startPos = endPos + 1;
97 endPos = str.find(delim, startPos);
98 ++i;
99 }
100 for (; i < N; ++i)
101 {
102 if (startPos > str.size() - 1)
103 result[i] = "";
104 else
105 {
106 result[i] = str.substr(startPos);
107 startPos = std::string::npos;
108 }
109 }
110 return result;
111 }

113 static inline auto SplitString(std::string_view str, char delim = ',')
114 {
115 std::vector<std::string_view> result;
116 size_t startPos = 0, endPos = str.find(delim);
117 while (endPos != std::string::npos)
118 {
119 result.push_back(str.substr(startPos, endPos - startPos));
120 startPos = endPos + 1;
121 endPos = str.find(delim, startPos);
122 }
123 result.push_back(str.substr(startPos));
124 return result;
125 }

541 static auto HandleHitObjects(std::ifstream& file)
542 {
543 std::string line;
544 std::vector<std::unique_ptr<HitObject>> objects;
545 while (details::GetLine(file, line))
546 {
547 auto [_, __, ___, typeStr] = details::SplitString<4>(line);
548 try
549 {
550 switch (HitObject::GetType(typeStr))
551 {
552 case Type::Circle:
553 objects.emplace_back(std::make_unique<Circle>(line));
554 break;
555 case Type::Slider:
556 objects.emplace_back(std::make_unique<Slider>(line));
557 break;
558 case Type::Spinner:
559 objects.emplace_back(std::make_unique<Spinner>(line));
560 break;
561 case Type::Hold:
562 objects.emplace_back(std::make_unique<Hold>(line));
563 break;
564 default:
565 throw std::exception{ "Hit object type not implemented" };
566 }
567 }
568 catch (...)
569 {
570 std::cerr << "Parsing Line: " << line << "failed!\n";
571 }
572 }
573 return objects;
574 }

579 /*
580 Some special bits representing hit object type
581 [type] is an 8-bit integer:
582 1 Circle
583 2 Slider
584 8 Spinner
585 *bit[2] set -> new combo
586 */
587 constexpr static inline unsigned CircleBit = 0b1;
588 constexpr static inline unsigned SliderBit = 0b1 << 1;
589 constexpr static inline unsigned ComboBit = 0b1 << 2;
590 constexpr static inline unsigned SpinnerBit = 0b1 << 3;
591 constexpr static inline unsigned HoldBit = 0b1 << 7;
592 Type type;

594 static inline Type GetType(int num)
595 {
596 if (num & CircleBit) return Type::Circle;
597 else if (num & SliderBit) return Type::Slider;
598 else if (num & SpinnerBit) return Type::Spinner;
599 else if (num & HoldBit) return Type::Hold;
600 }

603 {
