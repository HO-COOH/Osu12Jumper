123 static inline auto SplitString(std::string_view str, char delim = ',', bool appendRest = false)
124 {
125 std::array<std::string_view, N> result;
126 size_t startPos = 0, endPos = str.find(delim);
127 size_t i = 0;
128 while (endPos != std::string::npos && i < N)
129 {
130 if (appendRest && i + 1 == N)
131 result[i] = str.substr(startPos);
132 else
133 result[i] = str.substr(startPos, endPos - startPos);
134 startPos = endPos + 1;
135 endPos = str.find(delim, startPos);
136 ++i;
137 }
138 for (; i < N; ++i)
139 {
140 if (startPos > str.size() - 1)
141 result[i] = "";
142 else
143 {
144 result[i] = str.substr(startPos);
145 startPos = std::string::npos;
146 }
147 }
148 return result;

185 {

192 if (line.empty())

198 }
