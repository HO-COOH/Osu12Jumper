47 static inline auto SplitKeyVal(std::string_view line, char separator = ':')

98 template<size_t N>
99 static inline auto SplitString(std::string_view str, char delim = ',', bool appendRest = false)
100 {
101 std::array<std::string_view, N> result;
102 size_t startPos = 0, endPos = str.find(delim);
103 size_t i = 0;
104 while (endPos != std::string::npos && i < N)
105 {
106 if (appendRest && i + 1 == N)
107 result[i] = str.substr(startPos);
108 else
109 result[i] = str.substr(startPos, endPos - startPos);
110 startPos = endPos + 1;
111 endPos = str.find(delim, startPos);
112 ++i;
113 }
114 for (; i < N; ++i)
115 {
116 if (startPos > str.size() - 1)
117 result[i] = "";
118 else
119 {
120 result[i] = str.substr(startPos);
121 startPos = std::string::npos;
122 }
123 }
124 return result;
125 }

151 static inline auto GetLine(std::ifstream& file, std::string& line, bool indicateEmptyLine = true)
152 {
153 while (std::getline(file, line))
154 {
155 if (!line.empty() && line.back() == '\r')
156 line = line.substr(0, line.size() - 1);
157
158
159 if (line.empty())
160 {
161 if (indicateEmptyLine)
162 return false;
163 else
164 continue;
165 }
166
167 if (line[0] == '/' && line[1] == '/')
168 continue;
169
170 else if (auto const pos = line.find("//") != std::string::npos)
171 line = line.substr(0, pos);
172 return true;
173 }
174 return indicateEmptyLine ? false : static_cast<bool>(file);
175 }

784 static auto HandleHitObjects(std::ifstream& file)
785 {
786 std::string line;
787 std::vector<std::unique_ptr<HitObject>> objects;
788 while (details::GetLine(file, line))
789 {
790 auto [_, __, ___, typeStr] = details::SplitString<4>(line);
791 try
792 {
793 switch (HitObject::GetType(typeStr))
794 {
795 case Type::Circle:
796 objects.emplace_back(std::make_unique<Circle>(line));
797 break;
798 case Type::Slider:
799 objects.emplace_back(std::make_unique<Slider>(line));
800 break;
801 case Type::Spinner:
802 objects.emplace_back(std::make_unique<Spinner>(line));
803 break;
804 case Type::Hold:
805 objects.emplace_back(std::make_unique<Hold>(line));
806 break;
807 default:
808 throw std::runtime_error{ "Hit object type not implemented" };
809 }
810 }
811 catch (...)
812 {
813 std::cerr << "Parsing Line: " << line << "failed!\n";
814 }
815 }
816 return objects;
